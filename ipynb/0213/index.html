    test = SubhaloDataSet(_subhalo_id)

    x = test.DATA[0]["x"]
    y = test.DATA[0]["y"]
    m = test.DATA[0]["Masses"]
    hist, xed, yed = np.histogram2d(x, y, bins=101, weights=m)
