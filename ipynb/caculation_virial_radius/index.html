    def ChangeToCurrectCoordinates(x,y,z,cmx, cmy, cmz):
    HalfBoxSize = 35000/2
    if np.max(x) - np.min(x) > HalfBoxSize:
        x[x > HalfBoxSize] -= 35000
        if cmx > HalfBoxSize:
            cmx -= 35000
    if np.max(y) - np.min(y) > HalfBoxSize:
        y[y > HalfBoxSize] -= 35000
        if cmy > HalfBoxSize:
            cmy -= 35000
    if np.max(z) - np.min(z) > HalfBoxSize:
        z[z > HalfBoxSize] -= 35000
        if cmz > HalfBoxSize:
            cmz -= 35000
    return x,y,z,cmx, cmy, cmz

    def GetViri(SubhaloID):
    cmx, cmy, cmz = il.groupcat.loadSingle(basePath, 99, subhaloID=SubhaloID)["SubhaloCM"]

    x,y,z,m = GetAllData(SubhaloID)
    x,y,z,cmx, cmy, cmz = ChangeToCurrectCoordinates(x,y,z,cmx, cmy, cmz)

    r = np.sqrt((x - cmx) ** 2 + (y - cmy) ** 2 + (z - cmz) ** 2)

    R, m_t = np.sort(r), m[np.argsort(r)]
    cum_m = np.cumsum(m_t)

    _r = 2.1 * (cum_m * 10**10 / h / 10**15) ** (1 / 3) * (h / 0.7) ** (-2 / 3) * 10**3
    min_index = np.argmin((R / h - _r) ** 2)
    viri = (R/h)[min_index]
    return viri,cum_m[-1]/h
